def _format_overdue(minutes: float) -> str:
    """Format overdue minutes as human-readable string."""
    if minutes < 60:
        return f"{int(minutes)}m"
    hours = int(minutes // 60)
    mins = int(minutes % 60)
    if mins:
        return f"{hours}h {mins}m"
    return f"{hours}h"
