```python
class SpamWhitelist(SoftDeleteModel, TimestampedModel):
    class EntryType(models.TextChoices):
        EMAIL = "email", "Email"
        DOMAIN = "domain", "Domain"

    entry = models.CharField(max_length=255, db_index=True)
    entry_type = models.CharField(max_length=10, choices=EntryType.choices, default=EntryType.EMAIL)
    added_by = models.ForeignKey(settings.AUTH_USER_MODEL, on_delete=models.SET_NULL, null=True)
    reason = models.CharField(max_length=255, blank=True, default="")

    class Meta:
        ordering = ["-created_at"]
        unique_together = [("entry", "entry_type")]
```
