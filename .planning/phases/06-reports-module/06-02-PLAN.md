const centerTextPlugin = {
    id: 'centerText',
    beforeDraw(chart) {
        const { ctx, width, height } = chart;
        ctx.save();
        ctx.font = 'bold 28px "Plus Jakarta Sans"';
        ctx.fillStyle = '#1e293b';
        ctx.textAlign = 'center';

        ctx.fillText(chart.config._pct + '%', width / 2, height / 2);

};
