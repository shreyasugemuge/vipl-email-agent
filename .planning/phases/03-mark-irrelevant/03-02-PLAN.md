**D. JavaScript** -- add at the bottom of the template (inside a `<script>` block, or append to existing script):

```javascript
function openIrrelevantModal() {
    const modal = document.getElementById('irrelevant-modal');
    if (modal) {
        modal.classList.remove('hidden');
        const textarea = document.getElementById('irrelevant-reason-input');
        if (textarea) { textarea.value = ''; textarea.focus(); }
        const btn = document.getElementById('irrelevant-confirm-btn');
        if (btn) btn.disabled = true;
    }
}
function closeIrrelevantModal() {
    const modal = document.getElementById('irrelevant-modal');
    if (modal) modal.classList.add('hidden');
}
// Close on Escape
document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape') closeIrrelevantModal();
});
// Keyboard shortcut: I key opens modal (not in inputs)
document.addEventListener('keydown', function(e) {
    if ((e.key === 'I' || e.key === 'i') && !e.ctrlKey && !e.metaKey && !e.altKey) {
        if (['INPUT', 'TEXTAREA', 'SELECT'].includes(e.target.tagName)) return;
        if (document.getElementById('irrelevant-modal')) openIrrelevantModal();
    }
});
// Auto-open modal from context menu (check URL param)
if (new URLSearchParams(window.location.search).get('open_modal') === 'irrelevant') {
    setTimeout(function() { openIrrelevantModal(); }, 100);
}
// Also handle HTMX swap (for context menu -> detail panel load)
document.addEventListener('htmx:afterSettle', function(e) {
    if (e.detail.target && e.detail.target.id === 'thread-detail-panel') {
        if (new URLSearchParams(window.location.search).get('open_modal') === 'irrelevant') {
            setTimeout(function() { openIrrelevantModal(); }, 100);
        }
    }
});
```
