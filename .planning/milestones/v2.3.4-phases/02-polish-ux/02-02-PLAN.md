    Add to the `{% block extra_js %}` script block in email_list.html (after existing closeDetail and toggleFilters functions):

    1. **Keyboard navigation (UX-04):**
       ```javascript
       /* Keyboard navigation: Arrow keys between cards, Escape closes detail */
       document.addEventListener('keydown', function(e) {
           /* Skip when user is typing in form fields */
           if (['INPUT', 'SELECT', 'TEXTAREA'].includes(document.activeElement.tagName)) return;

           if (e.key === 'Escape') {
               closeDetail();
               return;
           }
           if (e.key === 'ArrowDown' || e.key === 'ArrowUp') {
               var cards = Array.from(document.querySelectorAll('#email-list [role="article"]'));
               if (!cards.length) return;
               var currentIdx = cards.indexOf(document.activeElement);
               var nextIdx;
               if (e.key === 'ArrowDown') {
                   nextIdx = currentIdx < cards.length - 1 ? currentIdx + 1 : 0;
               } else {
                   nextIdx = currentIdx > 0 ? currentIdx - 1 : cards.length - 1;
               }
               cards[nextIdx].focus();
               cards[nextIdx].scrollIntoView({ block: 'nearest' });
               e.preventDefault();
           }
       });
       ```

    2. **Loading skeleton (UX-05):**
       Add `htmx:beforeRequest` listener that injects skeleton HTML into `#detail-panel` ONLY when the target is `#detail-panel`:
       ```javascript
       /* Loading skeleton for detail panel */
       document.addEventListener('htmx:beforeRequest', function(e) {
           if (e.detail.target && e.detail.target.id === 'detail-panel') {
               e.detail.target.innerHTML = '...skeleton HTML...';
           }
       });
       ```

       Skeleton HTML should mimic the detail panel layout using Tailwind:

       - Use slate-100 and slate-200 grays per CONTEXT.md specifics
