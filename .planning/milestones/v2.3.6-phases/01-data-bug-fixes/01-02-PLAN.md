       ```javascript
       document.querySelectorAll('.toast-item').forEach(function(toast) {
           var startX = 0, currentX = 0, swiping = false;
           toast.addEventListener('touchstart', function(e) {
               startX = e.touches[0].clientX;
               currentX = startX;
               swiping = true;
           }, { passive: true });
           toast.addEventListener('touchmove', function(e) {
               if (!swiping) return;
               currentX = e.touches[0].clientX;
               var deltaX = currentX - startX;
               if (deltaX > 0) {  // Only swipe right
                   toast.style.transform = 'translateX(' + deltaX + 'px)';
                   toast.style.opacity = Math.max(0, 1 - deltaX / 200);
               }
           }, { passive: true });
           toast.addEventListener('touchend', function() {
               if (!swiping) return;
               swiping = false;
               if (currentX - startX > 50) {  // 50px threshold
                   toast.style.animation = 'toast-out 0.2s ease-in forwards';
                   setTimeout(function() { toast.remove(); }, 200);
               } else {
                   toast.style.transform = '';
                   toast.style.opacity = '';
               }
           });
       });
       ```
    4. Keep auto-dismiss after 4 seconds (already exists, no change).
