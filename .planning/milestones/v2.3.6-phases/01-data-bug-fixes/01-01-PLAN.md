def _parse_suggested_assignee(raw) -> dict:
    """Parse suggested_assignee from Claude response into structured dict."""
    if not raw:
        return {}
    if isinstance(raw, dict):
        return {
            "name": raw.get("name", ""),
            "reason": raw.get("reason", ""),
        }
    if isinstance(raw, str):
        if raw.strip():
            return {"name": raw.strip(), "reason": ""}
        return {}
    return {}
