       ```python
       @login_required
       @require_POST
       def bulk_assign(request):
           user = request.user
           # Use can_assign() permission helper (from Phase 1)
           is_admin = user.is_staff or user.role == User.Role.ADMIN
           if not is_admin:  # Phase 1 will have can_assign(); use is_admin pattern as fallback
               return HttpResponseForbidden("Permission denied.")

           thread_ids = request.POST.getlist("thread_ids")
           assignee_id = request.POST.get("assignee_id")
           if not thread_ids:
               return HttpResponse("Missing thread_ids", status=400)
           if not assignee_id:
               return HttpResponse("Missing assignee_id", status=400)

           assignee = get_object_or_404(User, pk=assignee_id, is_active=True)
           threads = Thread.objects.filter(pk__in=thread_ids, status__in=["new", "acknowledged"])

           previous_states = []
           assigned_pks = []
           for thread in threads:
               previous_states.append({
                   "thread_id": thread.pk,
                   "assigned_to_id": thread.assigned_to_id,
                   "status": thread.status,
               })
               thread.assigned_to = assignee
               thread.assigned_by = user
               if thread.status == "new":
                   thread.status = "acknowledged"
               thread.save(update_fields=["assigned_to", "assigned_by", "status", "updated_at"])
               ActivityLog.objects.create(
                   thread=thread,
                   email=thread.emails.order_by("-received_at").first(),
                   user=user,
                   action=ActivityLog.Action.ASSIGNED,
                   detail=f"Bulk assigned to {assignee.get_full_name() or assignee.username}",
               )
               assigned_pks.append(thread.pk)

           # Return updated thread list body (reuse thread_list logic)
           response = _render_thread_list_response(request)
           import json
           response["HX-Trigger"] = json.dumps({
               "showUndoToast": {
                   "message": f"Assigned {len(assigned_pks)} threads to {assignee.get_full_name() or assignee.username}",
                   "undo_url": reverse("emails:bulk_undo"),
                   "previous_states": previous_states,
                   "action_type": "assign",
               }
           })
           return response
       ```

       ```python
       @login_required
       @require_POST
       def bulk_mark_irrelevant(request):
           user = request.user
           is_admin = user.is_staff or user.role == User.Role.ADMIN
           if not is_admin:
               return HttpResponseForbidden("Permission denied.")

           thread_ids = request.POST.getlist("thread_ids")
           reason = (request.POST.get("reason") or "").strip()
           if not thread_ids:
               return HttpResponse("Missing thread_ids", status=400)
           if not reason:
               return HttpResponse("Reason is required", status=400)

           threads = Thread.objects.filter(pk__in=thread_ids).exclude(status="irrelevant")

           previous_states = []
           marked_pks = []
           for thread in threads:
               previous_states.append({
                   "thread_id": thread.pk,
                   "status": thread.status,
                   "assigned_to_id": thread.assigned_to_id,
               })
               thread.status = "irrelevant"
               thread.save(update_fields=["status", "updated_at"])
               ActivityLog.objects.create(
                   thread=thread,
                   email=thread.emails.order_by("-received_at").first(),
                   user=user,
                   action=ActivityLog.Action.CLOSED,
                   detail=f"Bulk marked irrelevant: {reason}",
               )
               marked_pks.append(thread.pk)

           response = _render_thread_list_response(request)
           import json
           response["HX-Trigger"] = json.dumps({
               "showUndoToast": {
                   "message": f"Marked {len(marked_pks)} threads as irrelevant",
                   "undo_url": reverse("emails:bulk_undo"),
                   "previous_states": previous_states,
                   "action_type": "mark_irrelevant",
               }
           })
           return response
       ```

    4. Add `bulk_undo` view:
       ```python
       @login_required
       @require_POST
       def bulk_undo(request):
           user = request.user
           is_admin = user.is_staff or user.role == User.Role.ADMIN
           if not is_admin:
               return HttpResponseForbidden("Permission denied.")

           import json
           try:
               previous_states = json.loads(request.POST.get("previous_states", "[]"))
           except (json.JSONDecodeError, TypeError):
               return HttpResponse("Invalid undo data", status=400)

           for state in previous_states:
               try:
                   thread = Thread.objects.get(pk=state["thread_id"])
                   thread.status = state.get("status", thread.status)
                   assigned_to_id = state.get("assigned_to_id")
                   thread.assigned_to_id = assigned_to_id
                   thread.save(update_fields=["status", "assigned_to", "updated_at"])
               except Thread.DoesNotExist:
                   continue

           return _render_thread_list_response(request)
       ```

          ```javascript
          // Bulk selection state management
          const bulkBar = document.getElementById('bulk-action-bar');
          const bulkCountEl = document.getElementById('bulk-count');

          function updateBulkState() {
              const checked = document.querySelectorAll('.thread-checkbox:checked');
              const count = checked.length;

              // Show/hide floating bar
              if (count > 0) {
                  bulkBar.classList.remove('translate-y-full', 'opacity-0', 'pointer-events-none');
                  bulkBar.classList.add('translate-y-0', 'opacity-100');
                  bulkBar.setAttribute('aria-label', `Bulk actions for ${count} selected threads`);
              } else {
                  bulkBar.classList.add('translate-y-full', 'opacity-0', 'pointer-events-none');
                  bulkBar.classList.remove('translate-y-0', 'opacity-100');
              }

              bulkCountEl.textContent = count + ' selected';

              // Update hidden inputs in both forms
              ['bulk-assign-thread-ids', 'bulk-irrelevant-thread-ids'].forEach(containerId => {
                  const container = document.getElementById(containerId);
                  if (container) {
                      container.innerHTML = '';
                      checked.forEach(cb => {
                          const input = document.createElement('input');
                          input.type = 'hidden';
                          input.name = 'thread_ids';
                          input.value = cb.value;
                          container.appendChild(input);
                      });
                  }
              });
          }

          function toggleSelectAll(masterCb) {
              document.querySelectorAll('.thread-checkbox').forEach(cb => {
                  cb.checked = masterCb.checked;
              });
              updateBulkState();
          }

          function clearBulkSelection() {
              document.querySelectorAll('.thread-checkbox').forEach(cb => { cb.checked = false; });
              const selectAll = document.getElementById('select-all-checkbox');
              if (selectAll) selectAll.checked = false;
              updateBulkState();
              hideIrrelevantMode();
          }

          function showIrrelevantMode() {
              document.getElementById('bulk-normal-mode').classList.add('hidden');
              document.getElementById('bulk-irrelevant-mode').classList.remove('hidden');
              document.getElementById('bulk-irrelevant-reason').focus();
          }

          function hideIrrelevantMode() {
              document.getElementById('bulk-normal-mode').classList.remove('hidden');
              document.getElementById('bulk-irrelevant-mode').classList.add('hidden');
              document.getElementById('bulk-irrelevant-reason').value = '';
              toggleIrrelevantConfirm();
          }

          function toggleIrrelevantConfirm() {
              const reason = document.getElementById('bulk-irrelevant-reason').value.trim();
              const btn = document.getElementById('bulk-irrelevant-confirm');
              btn.disabled = !reason;
          }

          // Escape key clears selection
          document.addEventListener('keydown', function(e) {
              if (e.key === 'Escape' && bulkBar && !bulkBar.classList.contains('pointer-events-none')) {
                  clearBulkSelection();
              }
          });

          // After HTMX swap, clear selection state
          document.body.addEventListener('htmx:afterSwap', function(e) {
              if (e.detail.target && e.detail.target.id === 'thread-list-body') {
                  clearBulkSelection();
              }
          });
          ```

       c. Add undo toast listener (extends existing toast system):
          ```javascript
          document.body.addEventListener('showUndoToast', function(e) {
              const data = e.detail;
              // Find the actual data -- HTMX wraps in different formats
              const payload = data.showUndoToast || data;

              const toast = document.createElement('div');
              toast.className = 'toast-item flex items-center gap-3 px-4 py-3 bg-white rounded-xl shadow-lg border border-slate-200 max-w-sm';
              toast.style.animation = 'toast-in 0.3s ease-out';

              const undoData = JSON.stringify(payload.previous_states || []);
              toast.innerHTML = `
                  <span class="text-sm text-slate-700">${payload.message}</span>
                  <button class="undo-btn text-sm font-semibold text-blue-600 hover:text-blue-800 cursor-pointer"
                          data-undo-url="${payload.undo_url}"
                          data-previous-states='${undoData}'>Undo</button>
                  <button onclick="this.closest('.toast-item').remove()"
                          class="text-slate-300 hover:text-slate-500 text-lg leading-none">&times;</button>
              `;

              // Wire undo button
              toast.querySelector('.undo-btn').addEventListener('click', function() {
                  const formData = new FormData();
                  formData.append('previous_states', this.dataset.previousStates);
                  formData.append('csrfmiddlewaretoken', document.querySelector('[name=csrfmiddlewaretoken]').value);
                  fetch(this.dataset.undoUrl, { method: 'POST', body: formData })
                      .then(r => r.text())
                      .then(html => {
                          document.getElementById('thread-list-body').innerHTML = html;
                          toast.remove();
                      });
              });

              const container = document.getElementById('toast-container');
              if (container) container.appendChild(toast);

              // Auto-dismiss after 10 seconds (longer than regular 4s toasts)
              setTimeout(() => {
                  toast.style.animation = 'toast-out 0.3s ease-in forwards';
                  setTimeout(() => toast.remove(), 300);
              }, 10000);
          });
          ```
